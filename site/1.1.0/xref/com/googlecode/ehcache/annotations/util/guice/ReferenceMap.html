36  /**
37   * Concurrent hash map that wraps keys and/or values in soft or weak
38   * references. Does not support null keys or values. Uses identity equality
39   * for weak and soft keys.
40   *
41   * <p>The concurrent semantics of {@link ConcurrentHashMap} combined with the
42   * fact that the garbage collector can asynchronously reclaim and clean up
43   * after keys and values at any time can lead to some racy semantics. For
44   * example, {@link #size()} returns an upper bound on the size, i.e. the actual
45   * size may be smaller in cases where the key or value has been reclaimed but
46   * the map entry has not been cleaned up yet.
47   *
48   * <p>Another example: If {@link #get(Object)} cannot find an existing entry
49   * for a key, it will try to create one. This operation is not atomic. One
50   * thread could {@link #put(Object, Object)} a value between the time another
51   * thread running {@code get()} checks for an entry and decides to create one.
52   * In this case, the newly created value will replace the put value in the
53   * map. Also, two threads running {@code get()} concurrently can potentially
54   * create duplicate values for a given key.
55   *
56   * <p>In other words, this class is great for caching but not atomicity.
57   *
58   * @author crazybob@google.com (Bob Lee)
59   */
60  @SuppressWarnings("unchecked")

71    /**
72     * Concurrent hash map that wraps keys and/or values based on specified
73     * reference types.
74     *
75     * @param keyReferenceType key reference type
76     * @param valueReferenceType value reference type
77     */
78    public ReferenceMap(ReferenceType keyReferenceType,
79        ReferenceType valueReferenceType,
80        ReferenceQueue<Object> referenceQueue) {
81      ensureNotNull(keyReferenceType, valueReferenceType, referenceQueue);
82  
83      if (keyReferenceType == ReferenceType.PHANTOM
84          || valueReferenceType == ReferenceType.PHANTOM) {
85        throw new IllegalArgumentException("Phantom references not supported.");
86      }
87  
88      this.delegate = new ConcurrentHashMap<Object, Object>();
89      this.keyReferenceType = keyReferenceType;
90      this.valueReferenceType = valueReferenceType;
91      this.referenceQueue = referenceQueue;
92    }

94    V internalGet(K key) {
95      Object valueReference = delegate.get(makeKeyReferenceAware(key));
96      return valueReference == null
97          ? null
98          : (V) dereferenceValue(valueReference);
99    }

106   V execute(Strategy strategy, K key, V value) {
107     ensureNotNull(key, value);
108     Object keyReference = referenceKey(key);
109     Object valueReference = strategy.execute(
110       this,
111       keyReference,
112       referenceValue(keyReference, value)
113     );
114     return valueReference == null ? null
115         : (V) dereferenceValue(valueReference);
116   }

119     return execute(putStrategy(), key, value);

217   public Set<Map.Entry<K, V>> entrySet() {
218     Set<Map.Entry<K, V>> entrySet = new HashSet<Map.Entry<K, V>>();
219     for (Map.Entry<Object, Object> entry : delegate.entrySet()) {
220       Map.Entry<K, V> dereferenced = dereferenceEntry(entry);
221       if (dereferenced != null) {

223       }
224     }
225     return Collections.unmodifiableSet(entrySet);

229    * Dereferences an entry. Returns null if the key or value has been gc'ed.
230    */
231   Entry dereferenceEntry(Map.Entry<Object, Object> entry) {
232     K key = dereferenceKey(entry.getKey());
233     V value = dereferenceValue(entry.getValue());
234     return (key == null || value == null)
235         ? null
236         : new Entry(key, value);

319    * {@code referenceType} and puts them in {@code out}. Returns
320    * {@code out}.

322   <T extends Collection<Object>> T dereferenceCollection(
323       ReferenceType referenceType, T in, T out) {
324     for (Object reference : in) {

339   /**
340    * Tests weak and soft references for identity equality. Compares references
341    * to other references and wrappers. If o is a reference, this returns true
342    * if r == o or if r and o reference the same non null object. If o is a
343    * wrapper, this returns true if r's referent is identical to the wrapped
344    * object.
345    */
346   static boolean referenceEquals(Reference r, Object o) {
347     // compare reference to reference.
348     if (o instanceof InternalReference) {
349       // are they the same reference? used in cleanup.
350       if (o == r) {
351         return true;
352       }
353 
354       // do they reference identical values? used in conditional puts.
355       Object referent = ((Reference) o).get();
356       return referent != null && referent == r.get();
357     }
358 
359     // is the wrapped object identical to the referent? used in lookups.
360     return ((ReferenceAwareWrapper) o).unwrap() == r.get();
361   }

566     }

569       return put(key, value);

573     public int hashCode() {
574       return key.hashCode() * 31 + value.hashCode();
575     }

577     @Override
578     public boolean equals(Object o) {
579       if (!(o instanceof ReferenceMap.Entry)) {
580         return false;
581       }

583       Entry entry = (Entry) o;
584       return key.equals(entry.key) && value.equals(entry.value);
585     }

589       return key + "=" + value;

607   private void writeObject(ObjectOutputStream out) throws IOException  {
608     out.defaultWriteObject();
609     out.writeInt(size());
610     for (Map.Entry<Object, Object> entry : delegate.entrySet()) {
611       Object key = dereferenceKey(entry.getKey());
612       Object value = dereferenceValue(entry.getValue());
613 
614       // don't persist gc'ed entries.
615       if (key != null && value != null) {
616         out.writeObject(key);
617         out.writeObject(value);
618       }
619     }
620     out.writeObject(null);
621   }

623   private void readObject(ObjectInputStream in) throws IOException,
624       ClassNotFoundException {
625     in.defaultReadObject();
626     int size = in.readInt();
627     this.delegate = new ConcurrentHashMap<Object, Object>(size);
628     while (true) {
629       K key = (K) in.readObject();
630       if (key == null) {
631         break;
632       }
633       V value = (V) in.readObject();
634       put(key, value);
635     }
636   }
